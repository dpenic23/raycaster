10  /**
11   * {@code AbstractRayTracerProducer} implements {@code IRayTracerProducer}
12   * interface and specifies object which are capable to create scene snapshots by
13   * using ray-tracing technique. This abstract class offers implemented method
14   * for producing such scenes, other objects extending this class has to
15   * implement only {@code calculateRGB(parameters)} method used for color
16   * calculations.
17   * 
18   * @author Domagoj
19   *
20   */
21  public abstract class AbstractRayTracerProducer implements IRayTracerProducer {
22  
23  	@Override
24  	public void produce(Point3D eye, Point3D view, Point3D viewUp,
25  			double horizontal, double vertical, int width, int height,
26  			long requestNo, IRayTracerResultObserver observer) {
27  
28  		System.out.println("Zapocinjem izracune...");
29  
30  		short[] red = new short[width * height];
31  		short[] green = new short[width * height];
32  		short[] blue = new short[width * height];
33  
34  		// normalized vector from eye position to view point
35  		Point3D og = view.sub(eye).normalize();
36  		// normalized view-up vector
37  		Point3D vuv = viewUp.normalize();
38  
39  		Point3D yAxis = vuv.sub(og.scalarMultiply(og.scalarProduct(vuv)))
40  				.normalize();
41  		Point3D xAxis = og.vectorProduct(yAxis).normalize();
42  
43  		Point3D screenCorner = view.sub(xAxis.scalarMultiply(horizontal / 2.0))
44  				.add(yAxis.scalarMultiply(vertical / 2.0));
45  
46  		Scene scene = RayTracerViewer.createPredefinedScene();
47  
48  		calculateRGB(red, green, blue, eye, xAxis, yAxis, screenCorner, scene,
49  				horizontal, vertical, width, height);
50  
51  		System.out.println("Izračuni gotovi...");
52  		observer.acceptResult(red, green, blue, requestNo);
53  		System.out.println("Dojava gotova...");
54  
55  	}

92  	/**
93  	 * Method which accepts arrays which represents intensity of red, blue and
94  	 * green color and calculates values for every pixel defined with other
95  	 * parameters. This method fills arrays only from screen height specified
96  	 * with y-minimal, to height specified as y-maximal.
97  	 * 
98  	 * @param red
99  	 *            Array of red intensities
100 	 * @param green
101 	 *            Array of green intensities
102 	 * @param blue
103 	 *            Array of blue intensities
104 	 * @param eye
105 	 *            Eye position vector
106 	 * @param xAxis
107 	 *            X axis vector
108 	 * @param yAxis
109 	 *            Y axis vector
110 	 * @param screenCorner
111 	 *            Screen corner defined by user view
112 	 * @param scene
113 	 *            Predefined scene
114 	 * @param horizontal
115 	 *            Horizontal width of observed space
116 	 * @param vertical
117 	 *            Vertical height of observed space
118 	 * @param width
119 	 *            Screen width
120 	 * @param height
121 	 *            Screen height
122 	 * @param yMin
123 	 *            Start height index for calculation, inclusively
124 	 * @param yMax
125 	 *            End height index for calculation, exclusively
126 	 */
127 	protected void calculateRGBFromTo(short[] red, short[] green, short[] blue,
128 			Point3D eye, Point3D xAxis, Point3D yAxis, Point3D screenCorner,
129 			Scene scene, double horizontal, double vertical, int width,
130 			int height, int yMin, int yMax) {
131 		short[] rgb = new short[3];
132 		int offset = yMin * width;
133 
134 		for (int y = yMin; y < yMax; y++) {
135 			for (int x = 0; x < width; x++) {
136 				Point3D xComponent = xAxis.scalarMultiply(x * horizontal
137 						/ (width - 1));
138 				Point3D yComponent = yAxis.scalarMultiply(y * vertical
139 						/ (height - 1));
140 				Point3D screenPoint = screenCorner.add(xComponent).sub(
141 						yComponent);
142 
143 				Ray ray = Ray.fromPoints(eye, screenPoint);
144 
145 				RayTracer.tracer(scene, ray, rgb);
146 
147 				red[offset] = rgb[0] > 255 ? 255 : rgb[0];
148 				green[offset] = rgb[1] > 255 ? 255 : rgb[1];
149 				blue[offset] = rgb[2] > 255 ? 255 : rgb[2];
150 
151 				offset++;
152 			}
153 		}
154 	}
